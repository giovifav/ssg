--- @classmod CollisionManager
-- Handles collision detection/resolution between moving entities and the level grid/entities.
-- Provides utility routines for broad/narrow phase checks used by ArrowManager and Game.
local CollisionManager = {}
CollisionManager.__index = CollisionManager

--- CollisionManager.new(level, gameWidth, gameHeight)
-- Creates a new CollisionManager object.
-- @param level Level The current active level
-- @param gameWidth number The width of the game world in grid units.
-- @param gameHeight number The height of the game world in grid units.
-- @return CollisionManager A new CollisionManager instance.
function CollisionManager.new(level, gameWidth, gameHeight)
    local self = setmetatable({}, CollisionManager)
    self.level = level
    self.gameWidth = gameWidth
    self.gameHeight = gameHeight
    return self
end

--- CollisionManager:getCollisionPixels(entity, moveDx, moveDy, collisionMode)
-- Determines the specific pixels to check for collision based on the specified collision mode.
-- @param entity Entity The moving entity (arrow).
-- @param moveDx number The intended movement in X direction.
-- @param moveDy number The intended movement in Y direction.
-- @param collisionMode string The collision mode to use ("pixel_perfect" or "aabb").
-- @return table A list of pixel data for collision checking.
function CollisionManager:getCollisionPixels(entity, moveDx, moveDy, collisionMode)
    local collisionPoints = {}

    if collisionMode == "pixel_perfect" then
        -- For pixel-perfect collision, check all pixels of the entity
        for _, pixel in ipairs(entity.pixels) do
            table.insert(collisionPoints, {
                x = math.floor(pixel.floatX + moveDx + 0.5),
                y = math.floor(pixel.floatY + moveDy + 0.5)
            })
        end
    elseif collisionMode == "aabb" then
        -- For AABB collision, use the bounding box corners
        local minX, minY = math.huge, math.huge
        local maxX, maxY = -math.huge, -math.huge

        for _, pixel in ipairs(entity.pixels) do
            minX = math.min(minX, pixel.floatX)
            minY = math.min(minY, pixel.floatY)
            maxX = math.max(maxX, pixel.floatX)
            maxY = math.max(maxY, pixel.floatY)
        end

        -- Calculate the corners of the bounding box in the next position
        local nextMinX = math.floor(minX + moveDx + 0.5)
        local nextMinY = math.floor(minY + moveDy + 0.5)
        local nextMaxX = math.floor(maxX + moveDx + 0.5)
        local nextMaxY = math.floor(maxY + moveDy + 0.5)

        -- Add the four corners of the bounding box
        table.insert(collisionPoints, {x = nextMinX, y = nextMinY})
        table.insert(collisionPoints, {x = nextMaxX, y = nextMinY})
        table.insert(collisionPoints, {x = nextMinX, y = nextMaxY})
        table.insert(collisionPoints, {x = nextMaxX, y = nextMaxY})
    else
        error("Invalid collisionMode: " .. tostring(collisionMode) .. ". Expected 'pixel_perfect' or 'aabb'.")
    end
    return collisionPoints
end

--- CollisionManager:checkPlayerMovement(playerEntity, dx, dy)
-- Checks if a player entity can move and handles collisions.
-- @param playerEntity Entity The player-controlled entity.
-- @param dx number The intended movement in the x-direction.
-- @param dy number The intended movement in the y-direction.
-- @return boolean True if the player can move, false otherwise.
-- @return boolean True if the player has won, false otherwise.
-- @return boolean True if the player has lost, false otherwise.
function CollisionManager:checkPlayerMovement(playerEntity, dx, dy)
    local playerCanMove = true
    local playerHasWon = false
    local playerHasLost = false
    local playerPixelsToMove = {}

    for _, pixel in ipairs(playerEntity.pixels) do
        local newFloatX = pixel.floatX + dx
        local newFloatY = pixel.floatY + dy
        
        local nextGridX = math.floor(newFloatX + 0.5)
        local nextGridY = math.floor(newFloatY + 0.5)

        table.insert(playerPixelsToMove, {newFloatX = newFloatX, newFloatY = newFloatY, nextGridX = nextGridX, nextGridY = nextGridY, original = pixel})

        if nextGridX < 0 or nextGridX >= self.gameWidth or nextGridY < 0 or nextGridY >= self.gameHeight then
            playerCanMove = false
            break
        end

        local collidedEntity = self.level.collisionMap[nextGridX .. "," .. nextGridY]
        if collidedEntity and collidedEntity ~= playerEntity then
            if collidedEntity.type == "wall" or collidedEntity.type == "player" then
                playerCanMove = false
                break
            elseif collidedEntity.type == "enemy" then
                playerHasLost = true
                break
            elseif collidedEntity.type == "trigger" and not collidedEntity.isMoving then
                self:resolveEntityCollision(collidedEntity, self.level)
                self.level.remainingTriggers = self.level.remainingTriggers - 1
            elseif collidedEntity.type == "collectible" then
                self:resolveEntityCollision(collidedEntity, self.level)
                self.level.remainingCollectibles = self.level.remainingCollectibles - 1
                if self.level.remainingCollectibles == 0 then
                    playerHasWon = true
                end
            end
        end
    end

    if playerCanMove then
        playerEntity:updateCollisionMapPosition(self.level, playerPixelsToMove)
    end

    return playerCanMove, playerHasWon, playerHasLost
end

--- CollisionManager:resolveEntityCollision(collidedEntity, level)
-- Resolves the collision by removing the collided entity from the level.
-- @param collidedEntity Entity The entity that was collided with.
-- @param level Level The current active level.
function CollisionManager:resolveEntityCollision(collidedEntity, level)
    for _, p in ipairs(collidedEntity.pixels) do
        level.collisionMap[p.x .. "," .. p.y] = nil
    end

    -- Remove from allEntities
    for i, entity in ipairs(level.allEntities) do
        if entity == collidedEntity then
            table.remove(level.allEntities, i)
            break
        end
    end

    -- Remove from specific entity lists
    if collidedEntity.type == "player" then
        for i, entity in ipairs(level.playerControlledEntities) do
            if entity == collidedEntity then
                table.remove(level.playerControlledEntities, i)
                break
            end
        end
    elseif collidedEntity.type == "player_arrow" then
        for i, entity in ipairs(level.playerArrows) do
            if entity == collidedEntity then
                table.remove(level.playerArrows, i)
                break
            end
        end
    else
        for i, entity in ipairs(level.entities) do
            if entity == collidedEntity then
                table.remove(level.entities, i)
                break
            end
        end
    end
end

return CollisionManager
