--- @classmod Game
-- Central game object: initialization, update loop, drawing, and input handling.
local Level = require("level")
local CollisionManager = require("collision_manager")
local Win = require("win")
local GameOver = require("gameover")

local Constants = require("constants")
local Utils = require("utils") -- Added for log function

local Game = {}
Game.__index = Game

--- Game.new()
-- Creates a new Game object and initializes game settings.
-- @return Game A new Game instance.
function Game.new()
    local self = setmetatable({}, Game)
    self.sceneManager = nil -- Will be set by main.lua after sceneManager is created

    -- Game settings
    self.gameWidth = Constants.GAME_WIDTH
    self.gameHeight = Constants.GAME_HEIGHT
    self.scale = Constants.GAME_SCALE
    self.playerSpeed = 15 -- pixels per second
    self.arrowSpeed = Constants.DEFAULT_ARROW_SPEED -- Using constant for arrow speed

    --- @type Level
    -- The current active level of the game.
    self.currentLevel = nil

    --- @type CollisionManager
    -- Manages collision detection and resolution.
    self.collisionManager = nil

    return self
end


--- Game:reset()
-- Resets the game to its initial playing state, reloading the level.
function Game:reset()
    self.currentLevel = Level.new(Constants.DEFAULT_LEVEL_PATH)
    self.currentLevel.arrowManager.arrowSpeed = self.arrowSpeed -- Set arrow speed
    self.collisionManager = CollisionManager.new(self.currentLevel, self.gameWidth, self.gameHeight)

    if #self.currentLevel.playerControlledEntities == 0 then
        Utils.log("Error: No player-controlled entities found in currentLevel.")
    else
        Utils.log("Player-controlled entities initialized successfully.")
    end
end

--- Game:update(dt)
-- Called every frame to update game logic.
-- Handles player movement, collision detection, entity updates, and win/loss conditions.
-- @param dt number The time elapsed since the last frame in seconds.
function Game:update(dt)
    if #self.currentLevel.playerControlledEntities == 0 then
        return
    end

    -- Player movement logic
    local dx, dy = 0, 0
    if love.keyboard.isDown('right') then
        dx = self.playerSpeed * dt
    end
    if love.keyboard.isDown('left') then
        dx = -self.playerSpeed * dt
    end
    if love.keyboard.isDown('up') then
        dy = -self.playerSpeed * dt
    end
    if love.keyboard.isDown('down') then
        dy = self.playerSpeed * dt
    end

    local playerHasWon = false
    local playerHasLost = false

    for _, playerEntity in ipairs(self.currentLevel.playerControlledEntities) do
        local canMove, won, lost = self.collisionManager:checkPlayerMovement(playerEntity, dx, dy)
        if not canMove then
            -- Player cannot move, but we still need to check for win/loss conditions from other players or arrows
        end
        if won then
            playerHasWon = true
        end
        if lost then
            playerHasLost = true
        end
    end

    self.currentLevel.arrowManager:update(dt)


    -- Check for game over from arrow collisions
    if self.currentLevel.gameOver then
        playerHasLost = true
    end

    -- Prioritize loss over win
    if playerHasLost then
        Utils.log("Hai perso!")
        self.sceneManager:set_scene(GameOver.new(self.sceneManager, self))
    elseif playerHasWon then
        Utils.log("Hai vinto!")
        self.sceneManager:set_scene(Win.new(self.sceneManager, self))
    end
end

--- Game:draw()
-- Called every frame to draw game elements.
-- Scales the graphics and draws the current level.
function Game:draw()
    love.graphics.scale(self.scale)
    self.currentLevel:draw()
end

--- Game:keypressed(key)
-- Handles keyboard key press events.
-- @param key string The name of the key pressed.
function Game:keypressed(key)
    if key == 'r' then
        self:reset()
    end
end

return Game
