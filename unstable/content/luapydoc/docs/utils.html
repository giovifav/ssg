--- @module utils
-- Provides utility functions for the game, including flood-fill for grouping pixels
-- and functions for detecting arrow shapes and their directions.
local Utils = {}
local Constants = require("constants")

--- Utils.floodFill(levelData, startX, startY, color, visited)
-- Groups adjacent pixels of the same color using a flood-fill algorithm.
-- This is used during level loading to identify and create entities from pixel art.
-- @param levelData ImageData The image data of the level.
-- @param startX number The starting X coordinate for the flood fill.
-- @param startY number The starting Y coordinate for the flood fill.
-- @param color table The RGB color to match for grouping pixels.
-- @param visited table A table used to keep track of visited pixels to prevent infinite loops.
-- @return table A table containing all grouped pixel coordinates ({x=x, y=y}).
function Utils.floodFill(levelData, startX, startY, color, visited)
    local group = {}
    local stack = {{
        x = startX,
        y = startY
    }}

    while #stack > 0 do
        local current = table.remove(stack)
        local x, y = current.x, current.y

        -- Check boundaries and visited status
        if x >= 0 and x < levelData:getWidth() and y >= 0 and y < levelData:getHeight() and not visited[x .. "," .. y] then
            local r, g, b = levelData:getPixel(x, y)
            if r == color.r and g == color.g and b == color.b then
                table.insert(group, {
                    x = x,
                    y = y
                })
                visited[x .. "," .. y] = true

                -- Add all 8 neighbors to the stack
                table.insert(stack, {
                    x = x + 1,
                    y = y
                })
                table.insert(stack, {
                    x = x - 1,
                    y = y
                })
                table.insert(stack, {
                    x = x,
                    y = y + 1
                })
                table.insert(stack, {
                    x = x,
                    y = y - 1
                })
                table.insert(stack, {
                    x = x + 1,
                    y = y + 1
                })
                table.insert(stack, {
                    x = x - 1,
                    y = y - 1
                })
                table.insert(stack, {
                    x = x + 1,
                    y = y - 1
                })
                table.insert(stack, {
                    x = x - 1,
                    y = y + 1
                })
            end
        end
    end
    return group
end


--- Utils.getAABB(entity, offsetX, offsetY)
-- Calculates the Axis-Aligned Bounding Box (AABB) for a given entity.
-- @param entity Entity The entity to calculate the AABB for.
-- @param offsetX number An offset to apply to the X coordinates.
-- @param offsetY number An offset to apply to the Y coordinates.
-- @return number minX The minimum X coordinate of the AABB.
-- @return number minY The minimum Y coordinate of the AABB.
-- @return number maxX The maximum X coordinate of the AABB.
-- @return number maxY The maximum Y coordinate of the AABB.
function Utils.getAABB(entity, offsetX, offsetY)
    offsetX = offsetX or 0
    offsetY = offsetY or 0
    local minX, minY = math.huge, math.huge
    local maxX, maxY = -math.huge, -math.huge

    for _, pixel in ipairs(entity.pixels) do
        minX = math.min(minX, pixel.floatX + offsetX)
        minY = math.min(minY, pixel.floatY + offsetY)
        maxX = math.max(maxX, pixel.floatX + offsetX)
        maxY = math.max(maxY, pixel.floatY + offsetY)
    end
    return minX, minY, maxX, maxY
end

--- Utils.getAABBRounded(entity, offsetX, offsetY)
-- Calculates the Axis-Aligned Bounding Box (AABB) for a given entity,
-- with coordinates rounded to the nearest integer for grid-based collision.
-- @param entity Entity The entity to calculate the AABB for.
-- @param offsetX number An offset to apply to the X coordinates.
-- @param offsetY number An offset to apply to the Y coordinates.
-- @return number minX The minimum X coordinate of the rounded AABB.
-- @return number minY The minimum Y coordinate of the rounded AABB.
-- @return number maxX The maximum X coordinate of the rounded AABB.
-- @return number maxY The maximum Y coordinate of the rounded AABB.
function Utils.getAABBRounded(entity, offsetX, offsetY)
    local minX, minY, maxX, maxY = Utils.getAABB(entity, offsetX, offsetY)
    return math.floor(minX), math.floor(minY), math.ceil(maxX), math.ceil(maxY)
end

--- Utils.log(message)
-- Conditionally prints a message to the console if DEBUG_MODE is enabled.
-- @param message string The message to print.
function Utils.log(message)
    if Constants.DEBUG_MODE then
        print(message)
    end
end

return Utils
