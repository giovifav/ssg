--- @classmod Entity
-- Defines the structure and behavior of game entities, such as players, enemies, walls, and collectibles.
-- @field type string The type of the entity (e.g., "player", "enemy", "wall", "collectible", "trigger").
-- @field pixels table A list of pixel data for the entity. Each pixel has `x`, `y` (integer grid coordinates),
--   `floatX`, and `floatY` (precise floating-point coordinates).
-- @field color table The RGB color of the entity.
-- @field isMoving boolean True if the entity is currently moving (e.g., an arrow), false otherwise.
-- @field direction string The current movement direction of the entity (e.g., "up", "down", "left", "right", "up-right", etc.).
local ArrowManager = require("arrow_manager")

local Entity = {}
Entity.__index = Entity

--- Entity.new(type, pixels, color)
-- Creates a new Entity object.
-- Initializes the entity's type, color, and sets up its pixels with both integer and floating-point positions.
-- @param type string The type of the entity.
-- @param pixels table A table of initial pixel coordinates ({x=x, y=y}).
-- @param color table The RGB color of the entity.
-- @param normalizedShape table The normalized pixel group representing the entity's shape (relative to 0,0).
-- @param bounds table The bounding box of the original pixel group.
-- @return Entity A new Entity instance.
function Entity.new(type, pixels, color, normalizedShape, bounds, initialDirection)
    local self = setmetatable({}, Entity)
    self.type = type
    self.color = color
    self.isMoving = false
    self.direction = initialDirection -- Set initial direction
    self.initialDirection = initialDirection -- Store original detected direction for shape transforms
    self.initialNormalizedShape = normalizedShape or {} -- Store the initial detected shape
    self.initialBounds = bounds or {minX=0, minY=0, maxX=0, maxY=0, width=0, height=0} -- Store initial bounds
    self.normalizedShape = normalizedShape or {} -- This will store the currently transformed shape
    self.bounds = bounds or {minX=0, minY=0, maxX=0, maxY=0, width=0, height=0} -- This will store the bounds of the currently transformed shape

    -- Initialize pixels with float positions
    self.pixels = {}
    for _, pixel in ipairs(pixels) do
        table.insert(self.pixels, {x = pixel.x, y = pixel.y, floatX = pixel.x, floatY = pixel.y})
    end
    return self
end

--- Entity:setMoving(direction)
-- Sets the entity's movement state and direction.
-- @param direction string The movement direction (e.g., "up", "down", "left", "right").
function Entity:setMoving(direction)
    self.isMoving = true
    self.direction = direction
end

--- Entity:updateCollisionMapPosition(level, newPixelsData)
-- Updates the entity's position in the collision map and its internal pixel coordinates.
-- @param level Level The current active level.
-- @param newPixelsData table A table of new pixel data, each with newFloatX, newFloatY, nextGridX, nextGridY, and original pixel reference.
function Entity:updateCollisionMapPosition(level, newPixelsData)
    -- Clear old positions from collision map
    for _, pixel in ipairs(self.pixels) do
        level.collisionMap[pixel.x .. "," .. pixel.y] = nil
    end

    -- Update pixel positions and add new positions to collision map
    for _, p in ipairs(newPixelsData) do
        p.original.floatX = p.newFloatX
        p.original.floatY = p.newFloatY
        p.original.x = p.nextGridX
        p.original.y = p.nextGridY
        level.collisionMap[p.nextGridX .. "," .. p.nextGridY] = self
    end
end

--- Entity:removeFromCollisionMap(level)
-- Removes the entity's current pixel positions from the collision map.
-- @param level Level The current active level.
function Entity:removeFromCollisionMap(level)
    for _, pixel in ipairs(self.pixels) do
        level.collisionMap[pixel.x .. "," .. pixel.y] = nil
    end
end

--- Entity:updateVisualShape()
-- Updates the entity's `pixels` table based on its `normalizedShape` and `direction`.
-- This method applies transformations (mirroring/rotation) to the base shape and keeps the shape centered.
function Entity:updateVisualShape()
    if not self.isMoving or not self.direction or not self.initialNormalizedShape or #self.initialNormalizedShape == 0 then
        return
    end

    -- Compute current bounding box center to preserve position after transform
    local minX, minY = math.huge, math.huge
    local maxX, maxY = -math.huge, -math.huge
    for _, pixel in ipairs(self.pixels) do
        minX = math.min(minX, pixel.x)
        minY = math.min(minY, pixel.y)
        maxX = math.max(maxX, pixel.x)
        maxY = math.max(maxY, pixel.y)
    end
    local currentCenterX = (minX + maxX) / 2
    local currentCenterY = (minY + maxY) / 2

    local transformedShape, transformedBounds = ArrowManager.getTransformedShape(
        self.initialNormalizedShape,
        self.initialBounds,
        self.initialDirection,
        self.direction
    )

    self.normalizedShape = transformedShape
    self.bounds = transformedBounds

    -- Center of the transformed shape in its own coordinate system (normalizedShape is relative to 0,0)
    local shapeCenterX = (self.bounds.width / 2 - 0.5)
    local shapeCenterY = (self.bounds.height / 2 - 0.5)

    -- Offset so that centers match
    local offsetX = currentCenterX - shapeCenterX
    local offsetY = currentCenterY - shapeCenterY

    local newPixels = {}
    for _, p in ipairs(self.normalizedShape) do
        local nx = math.floor(offsetX + p.x + 0.5)
        local ny = math.floor(offsetY + p.y + 0.5)
        table.insert(newPixels, { x = nx, y = ny, floatX = nx, floatY = ny })
    end
    self.pixels = newPixels
end

--- Entity:draw()
-- Draws the entity on the screen using its current floating-point pixel positions,
-- rounded to the nearest integer for rendering on the grid.
function Entity:draw()
    love.graphics.setColor(self.color.r, self.color.g, self.color.b)
    local points = {}
    for _, pixel in ipairs(self.pixels) do
        table.insert(points, math.floor(pixel.floatX + 0.5)) -- Round to nearest integer for drawing
        table.insert(points, math.floor(pixel.floatY + 0.5)) -- Round to nearest integer for drawing
    end
    love.graphics.points(points)

end

return Entity
